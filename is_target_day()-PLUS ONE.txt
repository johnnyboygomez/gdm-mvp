def is_target_day(participant_start_date):
    """Check if today is a target day OR target day + 1 for this participant"""
    today = date.today()
    delta_days = (today - participant_start_date).days
    
    # Original logic: target day
    is_exact_target = delta_days >= 7 and delta_days % 7 == 0
    
    # Grace period: target day + 1
    is_grace_day = delta_days >= 8 and (delta_days - 1) % 7 == 0
    
    return is_exact_target or is_grace_day
